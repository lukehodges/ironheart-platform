Test suite structure:

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { resourceService } from '../resources.service'
import * as repo from '../resources.repository'

vi.mock('../resources.repository')

const mockResourceRepo = vi.mocked(repo.resourceRepository)

describe('resourceService', () => {

  beforeEach(() => vi.clearAllMocks())

  describe('create', () => {
    it('creates a PERSON resource with correct defaults', async () => {
      const input = { type: 'PERSON' as const, name: 'Jane Smith' }
      const expected = { id: 'r-1', tenantId: 't-1', type: 'PERSON', name: 'Jane Smith', slug: 'jane-smith', capacity: 1, isActive: true, travelEnabled: false, userId: null, skillTags: null, homeAddressId: null, metadata: null, createdAt: new Date(), updatedAt: new Date() }
      mockResourceRepo.create.mockResolvedValue(expected)
      const result = await resourceService.create('t-1', input)
      expect(result.name).toBe('Jane Smith')
      expect(result.type).toBe('PERSON')
      expect(mockResourceRepo.create).toHaveBeenCalledWith('t-1', expect.objectContaining({ type: 'PERSON', name: 'Jane Smith' }))
    })

    it('throws on duplicate slug', async () => {
      mockResourceRepo.create.mockRejectedValue(new Error('unique constraint'))
      await expect(resourceService.create('t-1', { type: 'VEHICLE', name: 'Van 1' })).rejects.toThrow()
    })
  })

  describe('getById', () => {
    it('returns resource when found', async () => {
      const resource = { id: 'r-1', tenantId: 't-1', type: 'VEHICLE', name: 'Van 1', slug: 'van-1', capacity: 1, isActive: true, travelEnabled: true, userId: null, skillTags: ['driving'], homeAddressId: null, metadata: null, createdAt: new Date(), updatedAt: new Date() }
      mockResourceRepo.findById.mockResolvedValue(resource)
      const result = await resourceService.getById('t-1', 'r-1')
      expect(result).toEqual(resource)
    })

    it('throws NotFoundError when resource missing', async () => {
      mockResourceRepo.findById.mockResolvedValue(null)
      await expect(resourceService.getById('t-1', 'missing')).rejects.toMatchObject({ name: 'NotFoundError' })
    })
  })

  describe('list', () => {
    it('returns paginated resources', async () => {
      const items = [{ id: 'r-1' }, { id: 'r-2' }, { id: 'r-3' }]
      mockResourceRepo.list.mockResolvedValue({ rows: items.slice(0, 2), hasMore: true })
      const result = await resourceService.list('t-1', { limit: 2 })
      expect(result.rows).toHaveLength(2)
      expect(result.hasMore).toBe(true)
    })

    it('filters by type', async () => {
      mockResourceRepo.list.mockResolvedValue({ rows: [], hasMore: false })
      await resourceService.list('t-1', { type: 'VEHICLE' })
      expect(mockResourceRepo.list).toHaveBeenCalledWith('t-1', expect.objectContaining({ type: 'VEHICLE' }))
    })
  })

  describe('update', () => {
    it('updates resource and returns updated object', async () => {
      const updated = { id: 'r-1', tenantId: 't-1', type: 'PERSON', name: 'Updated Name', slug: 'updated-name', capacity: 1, isActive: true, travelEnabled: false, userId: null, skillTags: null, homeAddressId: null, metadata: null, createdAt: new Date(), updatedAt: new Date() }
      mockResourceRepo.update.mockResolvedValue(updated)
      const result = await resourceService.update('t-1', 'r-1', { name: 'Updated Name' })
      expect(result.name).toBe('Updated Name')
    })

    it('throws NotFoundError when resource not found for update', async () => {
      mockResourceRepo.update.mockResolvedValue(null)
      await expect(resourceService.update('t-1', 'missing', { name: 'x' })).rejects.toMatchObject({ name: 'NotFoundError' })
    })
  })

  describe('listAvailable', () => {
    it('returns resources not assigned to overlapping jobs', async () => {
      const available = [{ id: 'r-2', type: 'PERSON', name: 'Free Staff' }]
      mockResourceRepo.listAvailable.mockResolvedValue(available)
      const result = await resourceService.listAvailable('t-1', {
        date: new Date('2026-05-01'),
        startTime: '09:00',
        endTime: '11:00',
      })
      expect(result).toEqual(available)
    })

    it('filters available resources by skillTags', async () => {
      mockResourceRepo.listAvailable.mockResolvedValue([])
      await resourceService.listAvailable('t-1', { date: new Date('2026-05-01'), startTime: '09:00', endTime: '11:00', skillTags: ['gas-safe'] })
      expect(mockResourceRepo.listAvailable).toHaveBeenCalledWith('t-1', expect.objectContaining({ skillTags: ['gas-safe'] }))
    })
  })

  describe('delete', () => {
    it('deletes resource by id', async () => {
      mockResourceRepo.delete.mockResolvedValue(undefined)
      await expect(resourceService.delete('t-1', 'r-1')).resolves.not.toThrow()
    })
  })
})
```
